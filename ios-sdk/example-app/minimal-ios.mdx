## Minimal iOS example

```swift
import SwiftUI
import RampKit

@main
struct MyApp: App {
  init() {
    RampKit.initSdk(apiKey: "YOUR_API_KEY")
  }

  var body: some Scene {
    WindowGroup {
      ContentView()
    }
  }
}
```
