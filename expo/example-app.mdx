## Quick Start

```bash
# Clone the repository
git clone https://github.com/getrampkit/rampkit-expo
cd example-app

# Install dependencies
npm install

# Start development
npm start
```

## Key Code

### App.tsx

```typescript
import { useEffect, useState } from "react";
import { View, Button, Text, StyleSheet } from "react-native";
import { RampKit } from "rampkit-expo-dev";

export default function App() {
  const [ready, setReady] = useState(false);

  useEffect(() => {
    async function init() {
      await RampKit.init({
        appId: "YOUR_APP_ID",
        onOnboardingFinished: () => {
          console.log("Complete!");
        },
      });
      setReady(true);
    }
    init();
  }, []);

  return (
    <View style={styles.container}>
      <Text style={styles.title}>RampKit Demo</Text>
      <Button
        title="Start Onboarding"
        onPress={() => RampKit.showOnboarding()}
        disabled={!ready}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, justifyContent: "center", alignItems: "center" },
  title: { fontSize: 24, marginBottom: 20 },
});
```

## Run on Device

```bash
# iOS
npx expo run:ios

# Android
npx expo run:android
```

<Note>
Replace `YOUR_APP_ID` with your actual app ID from the RampKit dashboard.
</Note>
