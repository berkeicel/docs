A complete minimal example demonstrating the RampKit Expo SDK.

## App.tsx

```typescript
import { useEffect, useState } from "react";
import { View, Text, Button, StyleSheet } from "react-native";
import { RampKit } from "rampkit-expo-dev";

export default function App() {
  const [initialized, setInitialized] = useState(false);
  const [userId, setUserId] = useState<string | null>(null);

  useEffect(() => {
    async function initRampKit() {
      await RampKit.init({
        appId: "YOUR_APP_ID",
        environment: __DEV__ ? "development" : "production",
        autoShowOnboarding: false,
        onOnboardingFinished: (payload) => {
          console.log("Onboarding finished:", payload);
          // Navigate to main app content
        },
        onShowPaywall: (payload) => {
          console.log("Show paywall:", payload);
          // Integrate with RevenueCat or your paywall
        },
      });

      setInitialized(RampKit.isInitialized());
      setUserId(RampKit.getUserId());
    }

    initRampKit();

    // Cleanup on unmount
    return () => {
      RampKit.cleanup();
    };
  }, []);

  const handleShowOnboarding = () => {
    RampKit.showOnboarding();
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>RampKit Demo</Text>
      <Text style={styles.status}>
        Status: {initialized ? "✅ Initialized" : "⏳ Loading..."}
      </Text>
      {userId && (
        <Text style={styles.userId}>User ID: {userId.slice(0, 20)}...</Text>
      )}
      <View style={styles.buttonContainer}>
        <Button
          title="Show Onboarding"
          onPress={handleShowOnboarding}
          disabled={!initialized}
        />
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: "center",
    alignItems: "center",
    padding: 20,
    backgroundColor: "#f5f5f5",
  },
  title: {
    fontSize: 28,
    fontWeight: "bold",
    marginBottom: 20,
  },
  status: {
    fontSize: 16,
    marginBottom: 10,
  },
  userId: {
    fontSize: 12,
    color: "#666",
    marginBottom: 20,
  },
  buttonContainer: {
    marginTop: 20,
  },
});
```

## With RevenueCat Integration

```typescript
import { useEffect } from "react";
import { RampKit } from "rampkit-expo-dev";
import Purchases from "react-native-purchases";

export default function App() {
  useEffect(() => {
    async function init() {
      // Initialize RevenueCat
      await Purchases.configure({ apiKey: "YOUR_REVENUECAT_KEY" });

      // Initialize RampKit
      await RampKit.init({
        appId: "YOUR_APP_ID",
        onOnboardingFinished: async (payload) => {
          // Access collected variables
          console.log("User selections:", payload);
        },
        onShowPaywall: async (payload) => {
          // Show RevenueCat paywall
          try {
            const offerings = await Purchases.getOfferings();
            const pkg = offerings.current?.availablePackages[0];
            if (pkg) {
              await Purchases.purchasePackage(pkg);
              console.log("Purchase successful!");
            }
          } catch (error) {
            console.log("Purchase failed:", error);
          }
        },
      });
    }

    init();
  }, []);

  // ...
}
```

## Using Native APIs

```typescript
import { Haptics, StoreReview, Notifications } from "rampkit-expo-dev";

// Haptic feedback on button press
async function handleButtonPress() {
  await Haptics.impactAsync("medium");
  // Perform action...
}

// Request app store review
async function handleRateApp() {
  const available = await StoreReview.isAvailableAsync();
  if (available) {
    await StoreReview.requestReview();
  }
}

// Request notification permissions
async function handleEnableNotifications() {
  const result = await Notifications.requestPermissionsAsync({
    ios: {
      allowAlert: true,
      allowBadge: true,
      allowSound: true,
    },
  });
  
  console.log("Notifications granted:", result.granted);
  console.log("Status:", result.status);
}
```

## Custom Event Tracking

```typescript
import { RampKit } from "rampkit-expo-dev";

// Track screen views
function trackScreen(screenName: string) {
  RampKit.trackScreenView(screenName);
}

// Track button taps
function handleUpgradePress() {
  RampKit.trackCtaTap("upgrade_button", "Upgrade to Pro");
  // Navigate to upgrade flow...
}

// Track custom events
function trackPurchase(productId: string, amount: number) {
  RampKit.trackEvent("custom_purchase", {
    productId,
    amount,
    currency: "USD",
    timestamp: new Date().toISOString(),
  });
}
```

## Run the Example

```bash
# Clone the repository
git clone https://github.com/getrampkit/rampkit-expo

# Navigate to example
cd example-app

# Install dependencies
npm install

# Start Expo
npm start
```

<Tip>
Replace `YOUR_APP_ID` with your actual app ID from the RampKit dashboard. You can find it in Settings → App Configuration.
</Tip>
