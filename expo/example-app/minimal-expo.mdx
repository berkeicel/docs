A complete minimal example demonstrating the RampKit Expo SDK.

## App.tsx

```typescript
import { useEffect, useState } from "react";
import { View, Text, Button, StyleSheet } from "react-native";
import { RampKit } from "rampkit-expo-dev";

export default function App() {
  const [initialized, setInitialized] = useState(false);

  useEffect(() => {
    async function initRampKit() {
      await RampKit.init({
        appId: "YOUR_APP_ID",
        onOnboardingFinished: (payload) => {
          console.log("Onboarding finished:", payload);
        },
        onShowPaywall: (payload) => {
          console.log("Show paywall:", payload);
        },
      });

      setInitialized(true);
    }

    initRampKit();

    return () => {
      RampKit.cleanup();
    };
  }, []);

  return (
    <View style={styles.container}>
      <Text style={styles.title}>RampKit Demo</Text>
      <Text style={styles.status}>
        {initialized ? "✅ Ready" : "⏳ Loading..."}
      </Text>
      <Button
        title="Show Onboarding"
        onPress={() => RampKit.showOnboarding()}
        disabled={!initialized}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: "center",
    alignItems: "center",
    padding: 20,
  },
  title: {
    fontSize: 28,
    fontWeight: "bold",
    marginBottom: 20,
  },
  status: {
    fontSize: 16,
    marginBottom: 20,
  },
});
```

## With RevenueCat

```typescript
import { RampKit } from "rampkit-expo-dev";
import Purchases from "react-native-purchases";

useEffect(() => {
  async function init() {
    await Purchases.configure({ apiKey: "YOUR_REVENUECAT_KEY" });

    await RampKit.init({
      appId: "YOUR_APP_ID",
      onShowPaywall: async (payload) => {
        const offerings = await Purchases.getOfferings();
        // Show paywall UI
      },
    });
  }
  init();
}, []);
```

## Using Native APIs

```typescript
import { Haptics, StoreReview, Notifications } from "rampkit-expo-dev";

// Haptic feedback
await Haptics.impactAsync("medium");

// Request app store review
await StoreReview.requestReview();

// Request notification permissions
const result = await Notifications.requestPermissionsAsync();
```

## Custom Event Tracking

```typescript
// Track screen views
RampKit.trackScreenView("profile");

// Track button taps
RampKit.trackCtaTap("upgrade_button", "Upgrade to Pro");

// Track custom events
RampKit.trackEvent("custom_action", { key: "value" });
```

## Run the Example

```bash
git clone https://github.com/getrampkit/rampkit-expo
cd example-app
npm install
npm start
```

<Tip>
Replace `YOUR_APP_ID` with your app ID from the RampKit dashboard (Settings → App Configuration).
</Tip>
